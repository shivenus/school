Find the error in each of the following and explain how to correct it:

a
	char s[10] = "";
	strncpy(s, "hello", 5);
	printf("%s\n", s);

b
	printf("%s", 'a');

c
	char s[12] = "";
	strcpy(s, "Welcome Home");

d
	if (strcmp(string1, string2)) {
	puts("The strings are equal");
	}

---
a
	Error: Function strncpy does not write a terminating null character to array
s, because its third argument is equal to the length of the string "hello".
Correction: Make the third argument of strncpy 6, or assign '\0' to s[5].
	char s[10] = "";
	strncpy(s, "hello", 6);
	printf("%s\n", s);
	| OR
	char s[10] = "";
	strncpy(s, "hello", 5);
	s[5] = \0;
	printf("%s\n", s);

b
	needs %c
	printf("%c", a);

c
	needs bigger array for null character that will get added automatically
	char s[13] = "";
	strcpy(s, "Welcome Home");

d strcmp returns 0 when arguments are equal, in an if statement, that is equivalent to FALSE, so the condition needs to be either == 0 or ! (not true), here are some different ways to do it:
	if (strcmp(string1, string2) == 0) {
	puts("The strings are equal");
	}
	| OR
	if (!strcmp(string1, string2)) {
    puts("The strings are equal");
	}
	| OR
	if (strcmp(string1, string2)) {
    puts("The strings are not equal");
	}
	else {
		puts("The strings are equal");
	}
	| OR
	printf("%s\n", strcmp(string1, string2) ? "The strings are NOT equal" : "The strings are EQUAL"); 
	// basically, if the if statement is true/not 0, they're NOT equal, if it's false/0, they're EQUAL
